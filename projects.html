async function askFahadAI() { const inputField = document.getElementById('chat-input'); // تأكد أن هذا هو الـ ID لخانة الكتابة const message = inputField.value; if (!message) return; // أظهر رسالة المستخدم (يمكنك استخدام دالتك القديمة هنا للعرض) console.log("المستخدم قال: " + message); try { // إرسال السؤال للعقل في Hugging Face const response = await fetch("https://mohamedfahad-mohammed-ai.hf.space/run/predict", { method: "POST", headers: { "Content-Type": "application/json" }, body: JSON.stringify({ data: [message] }) }); const result = await response.json(); const aiReply = result.data[0]; // هنا تضع الكود الذي يعرض aiReply داخل فقاعة الدردشة في موقعك alert("ذكاء محمد فهد يقول: " + aiReply); // هذه للتجربة فقط، استبدلها بكود العرض الخاص بك } catch (error) { console.error("خطأ:", error); } inputField.value = ''; // مسح الخانة بعد الإرسال }
